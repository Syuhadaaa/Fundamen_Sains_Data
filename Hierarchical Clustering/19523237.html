#Single Linkage

df <- data.frame(States=rownames(USArrests), USArrests)
par(las=2)
par(mar=c(5,8,4,2))
df <- scale(df[, 2:5])
d <- dist(df, method = "euclidean")
clusters <- hclust(d, method = "single")
plot(clusters, cex = 0.6, hang = -1)

plot(clusters, cex = 0.6, hang = -1)
rect.hclust(clusters, k = 4, border = 2:5)
